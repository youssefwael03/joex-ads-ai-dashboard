function detectDirection(text: string): "rtl" | "ltr" {
  const arabicChars = (text.match(/[\u0600-\u06FF]/g) || []).length;
  const totalChars = text.replace(/\s/g, "").length;
  return totalChars > 0 && arabicChars / totalChars > 0.3
    ? "rtl" : "ltr";
}
